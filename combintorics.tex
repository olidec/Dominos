    \newcommand{\myplot}[5]{
     \begin{tikzpicture}
          \begin{axis}[
               grid = both,
               axis lines = middle,
               xlabel = $x$,
               x label style={anchor=west},
               ylabel = $y$,
               y label style={anchor=south},
               xmin = #2 , xmax = #3,
               ymin = #4, ymax = #5,
               restrict y to domain=#4:#5,
               xtick distance=1,
               ytick distance=1,
              ]
              \addplot[line width=1mm, domain=#2*1.1:#3*1.1, samples=100, color=red]{#1};
          \end{axis}
      \end{tikzpicture}
}
