    \newcommand{\hdomi}[2]{
        \begin{tikzpicture}
            \tikzmath{
                \dwidth = 13.5;
                \dheight = 4.5;
            }
        \draw[thick] (0,0) rectangle (\dwidth,\dheight);
        \draw (\dwidth/2,0) -- (\dwidth/2,\dheight);
        \node at (\dwidth/4,\dheight/2) {#1};
        \node at (3*\dwidth/4,\dheight/2) {#2};
        \end{tikzpicture}
    } 

    \newcommand{\vdomi}[2]{
        \begin{tikzpicture}
            \tikzmath{
                \dwidth = 8;
                \dheight = 3.5;
            }
        \draw[thick] (0,0) rectangle (\dwidth,\dheight);
        \draw (0,\dheight/2) -- (\dwidth,\dheight/2);
        \node at (\dwidth/2,3*\dheight/4) {\Large #1};
        \node at (\dwidth/2,\dheight/4) {\Large #2};
        \end{tikzpicture}
    }

    \newcommand{\myplot}[5]{
     \begin{tikzpicture}
          \begin{axis}[
               grid = both,
               axis lines = middle,
               xlabel = $x$,
               x label style={anchor=west},
               ylabel = $y$,
               y label style={anchor=south},
               xmin = #2 , xmax = #3,
               ymin = #4, ymax = #5,
               restrict y to domain=#4:#5,
               xtick distance=1,
               ytick distance=1,
              ]
              \addplot[line width=1mm, domain=#2*1.1:#3*1.1, samples=100, color=red]{#1};
          \end{axis}
      \end{tikzpicture}
}

\newcommand{\myvplot}[5]{
     \begin{tikzpicture}
          \begin{axis}[
               grid = both,
               axis lines = middle,
               xlabel = $x$,
               x label style={anchor=west},
               ylabel = $y$,
               y label style={anchor=south},
               xmin = #2 , xmax = #3,
               ymin = #4, ymax = #5,
               restrict y to domain=#4:#5,
               xtick distance=1,
               ytick distance=1,
              ]
              \addplot [line width=1mm, samples=100, color=red]coordinates {(#1,#5) (#1,#4)};
          \end{axis}
      \end{tikzpicture}
}
